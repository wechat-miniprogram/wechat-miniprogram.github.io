# KCells

KCells是列表分组，常用于嵌套一组 KCell 或者 KCheckbox.

# 使用指南

import {KCells} from 'kbone-ui'
1

# 实例

 <KCells title="带说明的列表项">
        <KCell
          value="标题文字"
          footer="说明文字" />
        <KCell
          title="带图标"
          footer="说明文字">
          <KIcon
            slot="icon"
            type="info" />
        </KCell>
    </KCells>
