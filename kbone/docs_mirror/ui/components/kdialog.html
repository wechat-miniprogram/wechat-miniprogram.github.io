# KDialog

弹窗组件。

# 使用指南

import { KDialog } from 'kbone-ui'
1

# 示例

<KDialog
  v-model="diaShow"
  :buttons="diaBtn"
  title="零度的田"
  desc="弹窗内容，告知当前状态、信息和解决方法，描述文字尽量控制在三行内"
/>

<script>
export default {
    data() {
        return {
            diaShow: false,
            diaBtn: [
                {
                    text: '确定',
                    click: () => {
                        console.log('确定')
                        this.diaShow = false
                    }
                }
            ],
        }
    }
}
</script>
